'use strict';

const doSomethingAsync = (something) => {
    return new Promise((resolve, reject) => {
        doSomething(something, (err, result) => {
            if (err) {
                reject(err);
            } else {
                resolve(result);
            }
        });
    });
};

doSomethingAsync('toto')
    .then((result) => console.log(result))
    .catch((err) => console.log(err));
